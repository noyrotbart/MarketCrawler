					arrowIcon.not('.glyphicon-home').removeClass("glyphicon-chevron-down").addClass("glyphicon-chevron-right");
					jQuery(this).find('> a span.glyphicon').removeClass("glyphicon-chevron-right").addClass("glyphicon-chevron-down");
					e.preventDefault();
				}
			});			
			
			/* Mobile - open/close menu */
			function closeMobileMenu() {
				menuButton.removeClass('active');
				mainMenu.removeAttr("style");
				firstVoice.removeClass('menu-selected');
				megaMenu.removeClass('mobile-menu');
				thirdLevel.removeClass('level-active');		
				jQuery('.mega span.glyphicon-chevron-down').removeClass("glyphicon-chevron-down").addClass("glyphicon-chevron-right");
			}
			
			function closeThirdLevel() {
				thirdLevel.removeClass('level-active');
				if (arrowIcon_2.hasClass('glyphicon-chevron-down')) {
					arrowIcon_2.removeClass("glyphicon-chevron-down").addClass("glyphicon-chevron-right");
				}
			}
			
			function addLayer() {
				jQuery('<div class="layerClose"/>').css({
					width: jQuery(document).width(),
					height: jQuery(document).height()
				}).appendTo('body');
			}
			
			menuButton.on('click', function(event) {
				jQuery('.mega span br').remove();
				addLayer();						
				
				mainMenu.slideToggle(function(){
					
					if (jQuery(this).is(':hidden')){
						closeMobileMenu();
						
					}else {
						menuButton.addClass('active');
						megaMenu.addClass('mobile-menu');
						
						/* Mobile - open third level*/
						jQuery('.mobile-menu .voce-2lev a').on('click', function(ev) {
							var secondLevel = jQuery(this).parents('.mega-open-list');
							
							if (secondLevel.next().hasClass('sub-mega-open-list')) {
								if (secondLevel.next().is(':hidden')) {
									closeThirdLevel();
									secondLevel.next().addClass('level-active');
									jQuery(this).find('span.glyphicon').removeClass("glyphicon-chevron-right").addClass("glyphicon-chevron-down");
									ev.preventDefault();
								}	
							}
						});
					}   
					
					jQuery('.layerClose').on('click',function(ev2) {
						closeMenu();
						jQuery(this).remove();
						ev2.stopPropagation();
					});
					
				});
				event.stopPropagation();
				event.preventDefault();
			});
			
			/* Mobile - close megamenu clicking on the arrow down */
			arrowIcon.on('click', function (ev) {
				if(jQuery(this).hasClass('glyphicon-chevron-down')) {
					firstVoice.removeClass("menu-selected");
					jQuery(this).removeClass("glyphicon-chevron-down").addClass("glyphicon-chevron-right");
					closeThirdLevel();
					ev.preventDefault();
					ev.stopPropagation();
				}
			});		
			
			/* Mobile - close third level clicking on the arrow down */
			arrowIcon_2.on('click', function (ev) {

					ev.stopPropagation();

			});					
			
			/* Close menu  (desktop and mobile) */
			function closeMenu(){

					mainMenu.slideToggle(function(){

					if (megaMenu.hasClass('mobile-menu')){
					
					}else{
						firstVoice.removeClass('menu-selected');
					}
				}

				closeMenu();
			});
			
			mainMenu.on('click',function(event){
				event.stopPropagation();
			});

var MONTH_NAMES=new Array('January','February','March','April','May','June','July','August','September','October','November','December','Jan','Feb','Mar','Apr','May','Jun','Jul','Aug','Sep','Oct','Nov','Dec');
var DAY_NAMES=new Array('Sunday','Monday','Tuesday','Wednesday','Thursday','Friday','Saturday','Sun','Mon','Tue','Wed','Thu','Fri','Sat');
function LZ(x) {return(x<0||x>9?"":"0")+x}

// ------------------------------------------------------------------
// formatDate (date_object, format)
// Returns a date in the output format specified.
// The format string uses the same abbreviations as in getDateFromFormat()
// ------------------------------------------------------------------
function formatDate(date,format) {
 format=format+"";
 var result="";
 var i_format=0;
 var c="";
 var token="";
 var y=date.getYear()+"";
 var M=date.getMonth()+1;
 var d=date.getDate();
 var E=date.getDay();
 var H=date.getHours();
 var m=date.getMinutes();
 var s=date.getSeconds();
 var yyyy,yy,MMM,MM,dd,hh,h,mm,ss,ampm,HH,H,KK,K,kk,k;
 // Convert real date parts into formatted versions
 var value=new Object();
 if (y.length < 4) {y=""+(y-0+1900);}
 value["y"]=""+y;
 value["yyyy"]=y;
 value["yy"]=y.substring(2,4);
 value["M"]=M;
 value["MM"]=LZ(M);
 value["MMM"]=MONTH_NAMES[M-1];
 value["NNN"]=MONTH_NAMES[M+11];
 value["d"]=d;
 value["dd"]=LZ(d);
 value["E"]=DAY_NAMES[E+7];
 value["EE"]=DAY_NAMES[E];
 value["H"]=H;
 value["HH"]=LZ(H);
 if (H==0){value["h"]=12;}
 else if (H>12){value["h"]=H-12;}
 else {value["h"]=H;}
 value["hh"]=LZ(value["h"]);
 if (H>11){value["K"]=H-12;} else {value["K"]=H;}
 value["k"]=H+1;
 value["KK"]=LZ(value["K"]);
 value["kk"]=LZ(value["k"]);
 if (H > 11) { value["a"]="PM"; }
 else { value["a"]="AM"; }
 value["m"]=m;
 value["mm"]=LZ(m);
 value["s"]=s;
 value["ss"]=LZ(s);
 while (i_format < format.length) {
  c=format.charAt(i_format);
  token="";
  while ((format.charAt(i_format)==c) && (i_format < format.length)) {
   token += format.charAt(i_format++);
   }
  if (value[token] != null) { result=result + value[token]; }
  else { result=result + token; }
  }
 return result;
 }

 function addRowToTable(tableBodyID, newsTitle, newsId, newsDate, urlDetail, urlPrefix) {
  var tbl = document.getElementById(tableBodyID);
  
  var lastRow = tbl.rows.length;
  var rowClass;
  if(lastRow % 2 == 0)
   rowClass = 'odd';
  else
   rowClass = 'even';
   
  // if there's no header row in the table, then iteration = lastRow + 1
  var iteration = lastRow;
  var row = tbl.insertRow(lastRow);
  row.className = rowClass;
  
  var cellLeft = row.insertCell(0);
  cellLeft.className = 'name';
  
  var dateFormatted = formatDate(newsDate,'dd NNN yy');
  var textNode = document.createTextNode(newsTitle);
  var dateTextNode = document.createTextNode('\u00a0\u00a0' +dateFormatted);
  
  if(urlPrefix) {
   var textPrefix = document.createTextNode(urlPrefix);
   cellLeft.appendChild(textPrefix);
  }
  
  if(urlDetail) {
   var anchorName = document.createElement("a");
   anchorName.setAttribute("href", urlDetail); 
   anchorName.appendChild(textNode);
   cellLeft.appendChild(anchorName);
  }
  else {
   cellLeft.appendChild(textNode);
  }
  cellLeft.appendChild(dateTextNode);
  return true;
